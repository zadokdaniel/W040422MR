              function flipACoin() {
                return new Promise((resolve, reject) => {
                  const randomNumber = Math.random();
              
                  if (randomNumber > 0.6) {
                    resolve("heads");
                  } else if (randomNumber > 0.2) {
                    resolve("tails");
                  } else {
                    reject("anomaly happened");
                  }
                });
              }
